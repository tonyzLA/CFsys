DROP TABLE inventory.Inventory;
CREATE TABLE inventory.Inventory
(ShelfCode VARCHAR(8),
 CategoryCode VARCHAR(8),
 PartNumber VARCHAR(15) NOT NULL,
 InterChgNumber1 VARCHAR(8),
 Description VARCHAR(255),
 PackDimension VARCHAR(20),
 Weight DECIMAL(10,2),
 SupplierCode VARCHAR(8),
 Quantity INTEGER DEFAULT 0, 
 Availablity CHAR(1),
 Remarks VARCHAR(255),
PRIMARY KEY (PartNumber),
FOREIGN KEY(PatchNo) REFERENCES PatchInfo(PatchNo),
FOREIGN KEY(CategoryCode) REFERENCES PartCategory(CategoryCode),
FOREIGN KEY(SupplierCode) REFERENCES Supplier(SupplierCode)
);
